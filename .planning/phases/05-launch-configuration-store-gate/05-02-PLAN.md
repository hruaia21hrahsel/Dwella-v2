**2. Create `components/UpdateGate.tsx`:**

This component wraps the app's content. It:
- No-ops when `!Updates.isEnabled` (development / Expo Go) — returns children immediately
- Uses `Updates.useUpdates()` to detect available updates
- When `isUpdateAvailable` is true, calls `Updates.fetchUpdateAsync()`
- When `isUpdatePending` is true, calls `Updates.reloadAsync()` to apply the update
- When `fetchUpdateAsync()` throws (fingerprint mismatch), shows a blocking forced-update screen
- The forced-update screen uses `Linking.openURL()` to redirect to the App Store or Play Store

```typescript
import { useEffect, useState } from 'react';
import { View, StyleSheet, Platform, Linking } from 'react-native';
import * as Updates from 'expo-updates';
import { Text, Button, Surface } from 'react-native-paper';
import { useTheme } from '@/lib/theme-context';

const STORE_URL = Platform.select({
  ios: 'https://apps.apple.com/app/dwella/id[APP_ID]',
  android: 'https://play.google.com/store/apps/details?id=com.dwella.app',
}) ?? '';

export function UpdateGate({ children }: { children: React.ReactNode }) {
  const [needsStoreUpdate, setNeedsStoreUpdate] = useState(false);
  const { colors } = useTheme();

  // Guard: no-op in dev / Expo Go where expo-updates is disabled
  if (!Updates.isEnabled) {
    return <>{children}</>;
  }

  const { isUpdateAvailable, isUpdatePending } = Updates.useUpdates();

  useEffect(() => {
    if (isUpdatePending) {
      Updates.reloadAsync();
    }
  }, [isUpdatePending]);

  useEffect(() => {
    if (isUpdateAvailable) {
      Updates.fetchUpdateAsync().catch(() => {
        // Fetch failed — runtime fingerprint mismatch, need store update
        setNeedsStoreUpdate(true);
      });
    }
  }, [isUpdateAvailable]);

  if (needsStoreUpdate) {
    return (
      <View style={[styles.screen, { backgroundColor: colors.background }]}>
        <Surface style={[styles.surface, { backgroundColor: colors.surface }]} elevation={4}>
          <Text variant="titleLarge" style={[styles.title, { color: colors.textPrimary }]}>
            Update Required
          </Text>
          <Text variant="bodyMedium" style={[styles.message, { color: colors.textSecondary }]}>
            A new version of Dwella is required to continue. Please update the app from the App Store or Play Store.
          </Text>
          <Button
            mode="contained"
            onPress={() => Linking.openURL(STORE_URL)}
            buttonColor={colors.primary}
          >
            Update App
          </Button>
        </Surface>
      </View>
    );
  }

  return <>{children}</>;
}

const styles = StyleSheet.create({
  screen: {
    flex: 1,
    justifyContent: 'center',
    padding: 24,
  },
  surface: {
    borderRadius: 16,
    padding: 24,
    gap: 16,
  },
  title: {
    fontWeight: '700',
  },
  message: {
    lineHeight: 22,
  },
});
```

IMPORTANT: The `STORE_URL` for iOS uses a placeholder `[APP_ID]` because the real App Store ID is not yet assigned. The Android URL uses the actual package name `com.dwella.app`. The iOS URL will be updated when the App Store listing is created (pre-launch checklist item).

IMPORTANT: The `if (!Updates.isEnabled)` guard MUST be placed before the `useUpdates()` hook call to avoid the hook running in dev mode. This means the hook is called conditionally — this is acceptable because the guard condition (`Updates.isEnabled`) never changes during the component's lifetime (it's determined at build time). If TypeScript or React linting flags this, wrap the hook usage in a separate inner component that is only rendered when `Updates.isEnabled` is true.

If the conditional hook pattern causes issues, restructure as:
```typescript
export function UpdateGate({ children }: { children: React.ReactNode }) {
  if (!Updates.isEnabled) return <>{children}</>;
  return <UpdateGateInner>{children}</UpdateGateInner>;
}

function UpdateGateInner({ children }: { children: React.ReactNode }) {
  // useUpdates() hook called unconditionally here
  const { isUpdateAvailable, isUpdatePending } = Updates.useUpdates();
  // ... rest of logic
}
```
