```typescript
/**
 * Validates that Claude's parsed JSON has all required fields for intent dispatch.
 * Prevents malformed AI output from reaching action handlers (EDGE-03).
 */
function isValidClaudeIntent(obj: unknown): obj is ClaudeIntent {
  if (typeof obj !== 'object' || obj === null) return false;
  const o = obj as Record<string, unknown>;
  return (
    typeof o.intent === 'string' &&
    typeof o.entities === 'object' && o.entities !== null &&
    typeof o.action_description === 'string' &&
    typeof o.needs_confirmation === 'boolean' &&
    typeof o.reply === 'string'
  );
}
```

**Step 2: Use the guard in callClaude()** — replace the try/catch block at the end of `callClaude()` (lines 503-513):

```typescript
  try {
    const parsed = JSON.parse(jsonStr.trim());
    if (!isValidClaudeIntent(parsed)) {
      // Fall back to general_chat — never execute DB action on invalid shape
      return {
        intent: 'general_chat',
        entities: {},
        action_description: 'general response',
        needs_confirmation: false,
        reply: typeof parsed?.reply === 'string' ? parsed.reply : rawContent,
      };
    }
    return parsed;
  } catch {
    return {
      intent: 'general_chat',
      entities: {},
      action_description: 'general response',
      needs_confirmation: false,
      reply: rawContent,
    };
  }
```

The key change: between `JSON.parse` succeeding and returning the result, we now validate the shape. If Claude returns valid JSON but missing required fields (e.g., `entities` is null or `intent` is missing), we fall back to `general_chat` instead of dispatching to an action handler with undefined fields.

```typescript
try {
```

```typescript
  });
}
```

This validates `filters.type` is one of `payments`, `tenants`, or `properties` before the query branch executes. If Claude returns an unexpected type, it falls back to `properties` search (the safest branch since it only reads user-owned properties with `is_archived = false`).
